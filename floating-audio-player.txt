<script>
const playerContainer = document.getElementById("floating-audio-player");
const audioPlayer = document.getElementById("audio-player");
let hideTimeout;

// Autoplay muted on load
window.addEventListener("load", () => {
  audioPlayer.play().catch(err => console.log("Autoplay blocked:", err));
});

// Unmute on first interaction
function unmuteAndPlay() {
  if(audioPlayer.muted){
    audioPlayer.muted = false;
    audioPlayer.play();
  }
}
["click","scroll","touchstart","keydown","mousemove"].forEach(evt =>
  document.addEventListener(evt, unmuteAndPlay, { once:true })
);

// Hide after 3s inactivity
function resetHideTimer() {
  playerContainer.style.opacity = "1";
  clearTimeout(hideTimeout);
  hideTimeout = setTimeout(() => {
    if(!playerContainer.matches(":hover")) {
      playerContainer.style.opacity = "0";
    }
  }, 3000);
}

// Show on interaction
["mousemove","keydown","scroll","click","touchstart"].forEach(evt =>
  document.addEventListener(evt, resetHideTimer)
);
playerContainer.addEventListener("mouseenter", () => {
  playerContainer.style.opacity = "1";
  clearTimeout(hideTimeout);
});
playerContainer.addEventListener("mouseleave", resetHideTimer);
resetHideTimer();
</script>
